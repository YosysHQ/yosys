# Single-port RAMs.

ram distributed $__ANALOGDEVICES_LUTRAM_SP_ {
	cost 1;
	option "ABITS" 5 {
		abits 5;
	}
	option "ABITS" 6 {
		abits 6;
	}
	width 1;
	init no_undef;
	prune_rom;
	port arsw "RW" {
		clock posedge;
	}
}
